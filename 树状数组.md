```cpp
#include<bits/stdc++.h>
using namespace std;
#define int long long
#define pii pair<int,int>
const int N=2e5+5;
int n,k,t[N];
int lowbit(int x) {return x&(-x);}
//单点修改
void update(int pos,int k)
{
    for(int i=pos;i<=2e5;i+=lowbit(i)) t[i]+=k;
}
//区间查询
int ask(int pos)
{
    int ans=0;
    for(int i=pos;i;i-=lowbit(i)) ans+=t[i];
    return ans;
}

int a[N],cnt=1,ans=0,q[N];
signed main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);cout.tie(0);
    cin>>n>>k;
    map<int,int>qs;
    for(int i=1;i<=n;i++) 
    {
        cin>>a[i];
        q[i]=q[i-1]+a[i];
        qs[q[i]]=1;
        qs[q[i]-k]=1;
    }
    qs[0]=1;
    for(auto i:qs) qs[i.first]=cnt++;
    update(qs[0],1);
    for(int i=1;i<=n;i++)
    {
        ans+=ask(qs[q[i]-k]);
        update(qs[q[i]],1);
    }
    cout<<ans<<"\n";
    return 0;
}
```

[题目2](https://www.luogu.com.cn/problem/P10235?contestId=159440)二分+树状数组。

```cpp
#include <bits/stdc++.h>
using namespace std;
#define int long long
#define pii pair<int,int>
const int N=2e5+5;
int a[N],n,k,t[N];

int lowbit(int x) {return x&(-x);}
//单点修改
void update(int pos,int k)
{
    for(int i=pos;i<=n;i+=lowbit(i)) t[i]+=k;
}
//区间查询
int ask(int pos)
{
    int ans=0;
    for(int i=pos;i;i-=lowbit(i)) ans+=t[i];
    return ans;
}
//离散化
struct node
{
    int val,ind;
    bool operator<(node &W)const
    {
        if(val==W.val) return ind<W.ind;
        else return val<W.val;
    }
}stu[N];
int ra[N];

bool reord(int mid)
{
    for(int i=1;i<=n;i++) t[i]=0;
    int ans=0,zu=0;
    vector<int>gai;
    for(int i=1;i<=n;i++)
    {
        int pos=ra[i];
        ans+=ask(n)-ask(pos);
        update(pos,1);
        if(ans>mid) 
        {
            ans=0;
            zu++;
            for(auto j:gai) update(j,-1);
            gai.clear();
        }
        gai.push_back(pos);
    }
    zu++;
    return zu<=k;
}

signed main() 
{
    ios::sync_with_stdio(false);
    cin.tie(0);cout.tie(0);
    int t;cin>>t;
    while(t--)
    {
        cin>>n>>k;
        for(int i=1;i<=n;i++) 
        {
            cin>>stu[i].val;//要求逆序对的数组
            stu[i].ind=i;
        }
        sort(stu+1,stu+n+1);
        for(int i=1;i<=n;i++) ra[stu[i].ind]=i;
        int l=0,r=n*n,ans=r;
        while(l<=r)
        {
            int mid=(l+r)/2;
            if(reord(mid)) ans=min(ans,mid),r=mid-1;
            else l=mid+1;
        }
        cout<<ans<<"\n";
    }
    return 0;
}
```
